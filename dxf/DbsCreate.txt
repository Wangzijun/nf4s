;Исходный файл для программы DbsCreate.exe

Format=circle    ; Задание формата создаваемого DBS-файла

File=DbsCreate.dbs       ; Имя создаваемого DBS-файла
; Если не указать, то запишется прямо в этот (текстовый) файл

Description=Произвольный текст...
; Всё, что написано здесь, будет помещено в новую запись 28

R=50	; Различные параметры

;=====================================
; Прямоугольник (со скошенными углами)
; Format=Rect
;
; Параметры
; W= Ширина
; H= Высота
; Дополнительные параметры
; UL.W= "ширина" скоса в левом верхнем углу
; UL.H= "высота" скоса в левом верхнем углу
; Аналогичные параметры LL.W и LL.H (для левого нижнего угла),
; UR.W и UR.H (правый верхний угол), LR.W и LR.H (правый нижний угол)


;=====================================
; Крест (со скруглениями)
; Format=Cross
;
; Параметры
; W= Полная ширина
; H= Полная высота
; UL.W= "ширина" (длина) левой верхней выемки
; UL.H= "высота" левой верхней выемки
; UL.R= радиус скругления в левой верхней выемке
; Аналогичные параметры LL.W, LL.H, LL.R (лев. нижн.),
; UR.W, UR.H, UR.R (прав. верхн.), LR.W, LR.H, LR.R (правый нижний угол)

;=====================================
; Треугольник
; Format=Triangle
;
; Параметры
; H= Высота
; L= Вынос левой (нижней) вершины
; R= Вынос правой (нижней) вершины

;=====================================
; Круг
; Format=Circle
;
; Параметры
; R= Радиус

;=====================================
; Кольцо
; Format=Ring
;
; Параметры
; R= Внешний радиус
; RI= Внутренний радиус

;=====================================
; Многоугольник
; Format=Polygon
;
; Параметры
; N= Количество сторон
; R= Радиус описанной окружности
; или (!)
; A= Длина стороны

;=====================================
; Сегмент круга
; Format=segment
;
; Параметры
; R= Радиус
; H= Высота

;---[Конец файла]---
